import java.util.*;

public class Solution {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = scanner.nextInt(), k = scanner.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) arr[i] = scanner.nextInt();

        Deque<Integer> minQ = new ArrayDeque<>();
        Deque<Integer> maxQ = new ArrayDeque<>();
        int left = 0, maxLen = 0;

        for (int right = 0; right < n; right++) {
            while (!minQ.isEmpty() && arr[right] < arr[minQ.peekLast()]) minQ.pollLast();
            while (!maxQ.isEmpty() && arr[right] > arr[maxQ.peekLast()]) maxQ.pollLast();
            minQ.addLast(right);
            maxQ.addLast(right);

            while (arr[maxQ.peekFirst()] - arr[minQ.peekFirst()] > k) {
                left++;
                if (minQ.peekFirst() < left) minQ.pollFirst();
                if (maxQ.peekFirst() < left) maxQ.pollFirst();
            }

            maxLen = Math.max(maxLen, right - left + 1);
        }

        System.out.println(maxLen);
        scanner.close();
    }
}
